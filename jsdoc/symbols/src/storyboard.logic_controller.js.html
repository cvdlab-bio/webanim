1 /**
  2  * @class Represents a controller to handle storyboard associated operations
  3  * @param {Listener} listener An object entitled to communicate to the UI
  4  */
  5 StoryboardController = function(listener) {
  6 	this.listener = listener;
  7 
  8 	//IDs management stuff
  9 	this.nextEventId = 1;
 10 	this.nextSegmentId = 1;
 11 	this.nextActorId = 1;
 12 
 13 	/**
 14 	 * Returns the id of the next event to be generated
 15 	 * @return {number} The id of the next event to be generated
 16 	 */
 17 	this.getEventId = function() {
 18 		return this.nextEventId++;
 19 	};
 20 
 21 	/**
 22 	 * Returns the id of the next segment to be generated
 23 	 * @return {number} The id of the next segment to be generated
 24 	 */
 25 	this.getSegmentId = function() {
 26 		return this.nextSegmentId++;
 27 	};
 28 
 29 	/**
 30 	 * Returns the id of the next actor to be generated
 31 	 * @return {number} The id of the next actor to be generated
 32 	 */
 33 	this.getActorId = function() {
 34 		return this.nextActorId++;
 35 	};
 36 
 37 	// Creation of a new Storyboard
 38 	this.storyboard = new Storyboard();
 39 	var source = new Event();
 40 	source.id = this.getEventId();
 41 	source.description = "Beginning of the animation.";
 42 	var sink = new Event();
 43 	sink.id = this.getEventId();
 44 	sink.description = "End of the animation.";
 45 	this.storyboard.addEvent(source);
 46 	this.storyboard.addEvent(sink);
 47 	this.storyboard.setSource(source);
 48 	this.storyboard.setSink(sink);
 49 
 50 	// Actors stuff
 51 	this.actors = [];
 52 
 53 	/**
 54 	 * Returns the actor with the specified id
 55 	 * @return {Actor} Actor with the specified id
 56 	 * @param {number} id The id of the actor to return
 57 	 */
 58 	this.getActorById = function(id) {
 59 		var actor;
 60 		var found = false;
 61 
 62 		for (var i = 0; i < this.actors.length && !found; i++) {
 63 			if (this.actors[i].id === id) {
 64 				found = true;
 65 				actor = this.actors[i];
 66 			};
 67 		};
 68 
 69 		return actor;
 70 	};
 71 };
 72 
 73 /* UC1: Add Actor. */
 74 
 75 /**
 76  * Adds an actor with the specified model and description properties to the controller
 77  * @param {String} description The description of the new actor to add
 78  * @param model The representation to associate to the new actor
 79  */
 80 StoryboardController.prototype.addActor = function(model, description) {
 81 	var newActor = new Actor();
 82 	newActor.id = this.getActorId();
 83 	newActor.model = model;
 84 	newActor.description = description;
 85 	this.actors.push(newActor);
 86 };

 90 /**
 91  * Adds a new event to the storyboard object of the controller
 92  * @param {String} description The description of the event to add
 93  */
 94 StoryboardController.prototype.addEvent = function(description) {
 95 	var newEvent = new Event();
 96 	newEvent.id = this.getEventId();
 97 	newEvent.description = description;
 98 	this.storyboard.addEvent(newEvent);
 99 };
100 
101 /* UC3: Add Segment. */
102 
103 /**
104  * Creates a new segment
105  * @param {number} fromId The id of the event the new segment departs from
106  * @param {number} toId The id of the event the new segment arrives to
107  */
108 StoryboardController.prototype.startAddSegment = function(fromId, toId) {
109 	var newSegment = new Segment();
110 	newSegment.id = this.getSegmentId();
111 	newSegment.from = this.storyboard.getEventById(fromId);
112 	newSegment.to = this.storyboard.getEventById(toId);
113 	this.newSegment = newSegment;
114 };
115 
116 /**
117  * Sets the description property of the new segment
118  * @param {String} description The description of the new segment
119  */
120 StoryboardController.prototype.setDescriptionForNewSegment = function(description) {
121 	this.newSegment.description = description;
122 };
123 
124 /**
125  * Sets the actor property of the new segment
126  * @param {number} actorId The id of the actor to associate the new segment to
127  */
128 StoryboardController.prototype.setActorForNewSegment = function(actorId) {
129 	var actor = this.getActorById(actorId);
130 	this.newSegment.actor = actor;
131 };
132 
133 /**
134  * Sets the behaviour property of the new segment
135  * @param {Function[]} behaviour The behaviour to associate to the new egment
136  */
137 StoryboardController.prototype.setBehaviourForNewSegment = function(behaviour) {
138 	this.newSegment.behaviour = behaviour;
139 };
140 
141 /**
142  * Sets the duration property of the new segment
143  * @param {number} duration The duration to associate to the new segment
144  */
145 StoryboardController.prototype.setDurationForNewSegment = function(duration) {
146 	this.newSegment.duration = duration;
147 };
148 
149 /**
150  * Adds the new segment to the storyboard object of the controller
151  */
152 StoryboardController.prototype.addSegment = function() {
153 	this.storyboard.addSegment(this.newSegment);
154 	this.newSegment = undefined;
155 };
156 
157 /* UC4: Process Storyboard. */
158 
159 /**
160  * Processes the storyboard object of the controller
161  */
162 StoryboardController.prototype.processStoryboard = function() {
163 	// Validity check
164 	if (!this.storyboard.isValid()) {
165 		this.listener.storyboardNotValid(this.storyboard.validityReport);
166 	} else {
167 		// Storyboard processing
168 		this.storyboard.executeCPM();
169 		this.storyboard.setStartTimeForSegments();
170 		this.listener.storyboardProcessingCompleted(this.storyboard);
171 	};
172 };

176 /**
177  * Removes the event with the specified id from the controller storyboard
178  * @param {number} eventId The id of the event to remove
179  */
180 StoryboardController.prototype.removeEvent = function(eventId) {
181 	var event = this.storyboard.getEventById(eventId);
182 	this.storyboard.removeEvent(event);
183 };
184 
185 /* UC6: Remove Segment. */
186 
187 /**
188  * Removes the segment with the specified id from the controller storyboard
189  * @param {number} segmentId The id of the segment to remove
190  */
191 StoryboardController.prototype.removeSegment = function(segmentId) {
192 	var segment = this.storyboard.getSegmentById(segmentId);
193 	this.storyboard.removeSegment(segment);
194 };

196 /**
197  * Returns the structured data the timeline might needs to show the segments of the actors
198  * @return {Object[]} Structured data of the segments associated to the actor
199  */
200 StoryboardController.prototype.populateTimeline = function() {
201 	return this.storyboard.actor2SegmentsData(this.actors);
202 }
