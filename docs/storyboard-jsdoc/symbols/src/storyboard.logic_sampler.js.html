1 /**
  2  * A Sampler extracts samples of an
  3  * Actor's configurations during its
  4  * Segments.
  5  * @param {function} a2s The Storyboard actor2Segments function, to retrieve the Actors' Segments.
  6  */
  7 Sampler = function(a2s) {
  8 	this.a2s = a2s;
  9 };
 10 
 11 /**
 12  * Returns an Actor's configuration sample at a given time.
 13  * @param {number} t Time of the required sample
 14  * @return {function} behaviour The behaviour associated to the Actor's Segment
 15  */
 16 Sampler.prototype.sample = function(actor, t) {
 17 	//  Get Actor's Segments
 18 	var segments = this.a2s(actor);
 19 
 20 	// Search Segment containing t
 21 	var segment;
 22 	var found = false;
 23 	for (var i = 0; i < segments.length && !found; i++) {
 24 		if (segments[i].tStart <= t && segments[i].tStart + segments[i].duration >= t) {
 25 			segment = segments[i];
 26 			found = true;
 27 		};
 28 	};
 29 
 30 	// Return configuration in t
 31 	var p = (t - segment.tStart) / segment.duration;
 32 	return segment.behaviour(p);
 33 };
 34 
 35 /**
 36  * Returns Actor's configuration samples from tStart
 37  * to tStop, given a framereate (in frames per
 38  * second).
 39  * @param {Actor} actor The specified actor
 40  * @param {number} tStart Starting time
 41  * @param {number} tStop Ending time
 42  * @param {number} fps Framerate, frames per second
 43  * @return {Function()} The Actor's configuration samples
 44  */
 45 Sampler.prototype.samples = function(actor, tStart, tStop, fps) {
 46 	//  Get Actor's Segments
 47 	var segments = this.a2s(actor);
 48 
 49 	// Search Segments from tStart to tStop
 50 	var segmentsToSample = [];
 51 	for (var i = 0; i < segments.length && segments[i].tStart <= tStop; i++) {
 52 		if (segments[i].tStart + segments[i].duration >= tStart) {
 53 			segmentsToSample.push(segments[i]);
 54 		};
 55 	};
 56 
 57 	// Get samples
 58 	var samples = [];
 59 	var delta = 1 / fps;
 60 	for (var i = 0; i < segmentsToSample.length; i++) {
 61 		var s = segmentsToSample[i];
 62 
 63 		// Dealing with truncated Segments
 64 		var t1 = s.tStart;
 65 		var t2 = s.tStart + s.duration;
 66 		if (s.tStart < tStart) {
 67 			t1 = tStart;
 68 		};
 69 		if (s.tStart + s.duration > tStop) {
 70 			t2 = tStop;
 71 		};
 72 
 73 		var t = t1;
 74 		while (t < t2) {
 75 			var p = (t - s.tStart) / s.duration;
 76 			samples.push(s.behaviour(p));
 77 			t += delta;
 78 		};
 79 	};
 80 
 81 	// Return samples
 82 	return samples;
 83 };
