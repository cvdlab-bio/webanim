  2  *	@class Represents an object of type Event, a node of the storyboard graph

  4 Event = function() {
  5 	/**
  6 	 *	Array of incoming segments
  7 	 *	@type Segment[]
  8 	 */
  9 	this.incomingSegments = [];
 10 	/**
 11 	 *	Array of outgoing segments
 12 	 *	@type Segment[]
 13 	 */
 14 	this.outgoingSegments = [];
 15 	/**
 16 	 *	Boolean to check if a node has been visited
 17 	 *	@type Boolean
 18 	 */
 19 	this.marked = false;
 20 };

 22 /**
 23  *	Returns the indegree value of the event
 24  *	@return {number} The indegree value
 25  */
 26 Event.prototype.inDegree = function() {
 27 	return this.incomingSegments.length;
 28 };

 30 /**
 31  *	Returns the outdegree value of the event
 32  *	@return {number} The outdegree value
 33  */
 34 Event.prototype.outDegree = function() {

 36 };

 83 /**
 84  *	@class Represents an object of type Storyboard, a graph
 85 */
 86 Storyboard = function() {
 87 
 88 	/**
 89 	 *	Array of events composing the storyboard graph

 91 	 */
 92 	this.events = [];

 94 	 *	Array of segments composing the storyboard graph

 96 	 */

116 /**
117  *	Returns an event of the storyboard graph with the specified id
118  *	@param {number} id The id of the event to return
119  *  @return {Event} The event of the storyboard with the specified id
120  */
121 Storyboard.prototype.getEventById = function(id) {
122 	var event;
123 	var found = false;
124 
125 	for (var i = 0; i < this.events.length && !found; i++) {
126 		if (this.events[i].id === id) {
127 			found = true;
128 			event = this.events[i];
129 		};
130 	};
131 
132 	return event;
133 };

163 /**
164  *	Adds a segment to the storyboard graph
165  *	@param {Segment} segment The segment to add to the storyboard graph
166  */
167 Storyboard.prototype.addSegment = function(segment) {
168 	var from = segment.from;
169 	var to = segment.to;
170 
171 	from.addOutgoingSegment(segment);
172 	to.addIncomingSegment(segment);
173 
174 	this.segments.push(segment);
175 	this.topologicallySorted = false;
176 };

178 /**
179  *	Removes an event from the storyboard graph, causing all the related segments to be removed too
180  *	@param {Event} event The event to remove from the storyboard graph
181  */
182 Storyboard.prototype.removeEvent = function(event) {
183 	// Remove all related Segments
184 	var segmentsToRemove = this.segments.filter(function(segment) {
185 		return segment.from === event || segment.to === event;
186 	});
187 	for (var i = 0; i < segmentsToRemove.length; i++) {
188 		this.removeSegment(segmentsToRemove[i]);
189 	};
190 
191 	// Remove the Event
192 	var i = this.events.indexOf(event);
193 	this.events.splice(i,1);
194 
195 	this.topologicallySorted = false;
196 };
197 
198 /**
199  *	Removes a segment from the storyboard graph, updating all the related events
200  *	@param {Segment} segment The segment to remove from the storyboard graph
201  */
202 Storyboard.prototype.removeSegment = function(segment) {
203 	// Update related Events
204 	segment.from.removeOutgoingSegment(segment);
205 	segment.to.removeIncomingSegment(segment);
206 
207 	// Remove the Segment
208 	var i = this.segments.indexOf(segment);
209 	this.segments.splice(i,1);
210 
211 	this.topologicallySorted = false;
212 };

214 /**
215  * Returns true if and only if the events in the Storyboard have
216  * correct degrees (indegree = 0 for source, outdegree = 0
217  * for sink, indegree != 0 and outdegree != 0 for remaining
218  * events).
219  *
220  * It also fills in the 'validityReport' property when it encounters an event
221  * which doesn't follow the rule mentioned before. Its properties are:
222  *
223  * - incoming: true if and only if the value of the indegree of an event other than the source is 0.
224  * - outgoing: true if and only if the value of the outdegree of an event other than the sink is 0.
225  * - source: true iff the indegree of the source event is different from 0.
226  * - sink: true iff the outdregree of the sink event is different from 0.
227  * - event: the event with wrong degree value.
228  *
229  * @return {Boolean} Checks if all the events of the storyobard have correct degrees 
230  *
231  */
232 Storyboard.prototype.hasCorrectDegrees = function() {
233 	var correct = true;
234 	for (var i = 0; i < this.events.length && correct; i++) {
235 		var e = this.events[i];
236 		if (e === this.source) {
237 			if (e.inDegree() !== 0) {
238 				this.validityReport.degrees.error = true;
239 				this.validityReport.degrees.source = true;
240 				correct = false;
241 			};
242 		} else if (e === this.sink) {
243 			if (e.outDegree() !== 0) {
244 				this.validityReport.degrees.error = true;
245 				this.validityReport.degrees.sink = true;
246 				correct = false;
247 			};
248 		} else {
249 			if (e.inDegree() === 0) {
250 				this.validityReport.degrees.error = true;
251 				this.validityReport.degrees.incoming = true;
252 				this.validityReport.degrees.event = e;
253 				correct = false;
254 			};
255 			if (e.outDegree() === 0) {
256 				this.validityReport.degrees.error = true;
257 				this.validityReport.degrees.outgoing = true;
258 				this.validityReport.degrees.event = e;
259 				correct = false;
260 			};
261 		};
262 	};
263 	return correct;
264 };

266 /**
267  * For simplicity and/or performances, some
268  * graph algorithms mark nodes instead of
269  * copying and deleting them. This function removes these
270  * marks, thus it is called by each function that needs to use them.
271  */
272 Storyboard.prototype.resetMarks = function() {
273 	this.events.forEach(function(event) {
274 		if (event.marked) event.marked = false;
275 	});
276 };

278 /**
279  * All events must be reachable from source event.
280  * This can be checked by using any non-cycling
281  * algorithm for graph visit and marking visited
282  * events: the storyboard is valid if and only if every event
283  * is marked.
284  *
285  * @return {Boolean} Checks if all the events of the storyboard are reachable from the source
286  */
287 Storyboard.prototype.hasAllEventsReachable = function() {
288 	// Graph visit for marking Events
289 	this.resetMarks();
290 	var eventsToVisit = [ this.source ];
291 	while (eventsToVisit.length !== 0) {
292 		var next = eventsToVisit.pop();
293 		if (!next.marked) {
294 			next.marked = true;
295 			next.outgoingSegments.forEach(function(segment) {
296 				if (!segment.to.marked) {
297 					eventsToVisit.push(segment.to);
298 				};
299 			});
300 		};
301 	};
302 
303 	// Search for non-marked Events
304 	var nonMarked = false;
305 	for (var i = 0; i < this.events.length && !nonMarked; i++) {
306 		if (!this.events[i].marked) nonMarked = true;
307 	};
308 
309 	// Filling validity report
310 	if (nonMarked) {
311 		this.validityReport.reachability = true;
312 	};
313 
314 	// Finally, return check result
315 	return !nonMarked;
316 };

318 /**
319  * Resets the topological order of the events of the storyboard
320  */
321 Storyboard.prototype.resetTopologicalOrder = function() {
322 	this.topologicallySorted = false;
323 	this.events.forEach(function(event) {
324 		if (event.topologicalOrder !== undefined) {
325 			event.topologicalOrder = undefined;
326 		};
327 	});

330 /**
331  * Kahn's algorithm for topological sorting.
332  * It is also suitable for detecting cycles.
333  * @return {Boolean} Checks if the storyboard graph contains cycles.
334  */
335 Storyboard.prototype.hasCycles = function() {
336 	// Resetting...
337 	this.resetMarks();
338 	this.resetTopologicalOrder();
339 
340 	// Kahn's algorithm
341 	var S = [ this.source ];
342 	var k = 0;
343 	while (S.length != 0) {
344 		// Remove an Event from S
345 		var event = S.pop();
346 		// Number the chosen Event
347 		event.topologicalOrder = k;
348 		k++;
349 
350 		// For each outgoing Segment ...
351 		event.outgoingSegments.forEach(
352 			function(segment) {
353 				// ... mark it
354 				segment.marked = true;
355 
356 				// ... push its TO Event into S iff it has no marked incoming Segments
357 				var to = segment.to;
358 				var incomings = to.incomingSegments;
359 				var found = false;
360 				for (var i = 0; i < incomings.length && !found; i++) {
361 					if (!incomings[i].marked) found = true;
362 				};
363 				if (!found) S.push(to);
364 			}
365 		);
366 	};
367 
368 	// Search for non-marked Segments
369 	var found = false;
370 	for (var i = 0; i < this.segments.length && !found; i++) {
371 		if (!this.segments[i].marked) found = true;
372 	};
373 
374 	// Iff there are non-marked Segments, the Storyboard contains cycles
375 	if (found) {
376 		this.validityReport.cycles = true;
377 		return true;
378 	} else {
379 		this.events.sort(function(e1, e2) {
380 			return e1.topologicalOrder - e2.topologicalOrder;
381 		});
382 		this.topologicallySorted = true;
383 		return false;
384 	};
385 };

395 /**
396  * Performs a full validation of the Storyboard.
397  * @return {Boolean} Checks if the storyboard graph is valid.
398  */
399 Storyboard.prototype.isValid = function() {
400 	this.resetValidityReport();
401 
402 	if (this.hasCorrectDegrees()) {
403 		if (this.hasAllEventsReachable()) {
404 			if (!this.hasCycles()) return true;
405 		};
406 	};
407 
408 	return false;
409 };

411 /**
412  * Detects cycles and provides topological sorting. It's necessary to the Critical Path Method.
413  * @return {Boolean} Checks if the topologically sorted storyboard graph contains cycles.
414  */
415 Storyboard.prototype.topologicalSort = function() {
416 	if (!this.topologicallySorted) {
417 		// Cycles detection also provides
418 		// topological sorting
419 		this.hasCycles();
420 	};
421 };

423 /**
424  * Cleans data before storyboard processing.
425  */
426 Storyboard.prototype.resetTimes = function() {
427 	this.events.forEach(function(event) {
428 		event.tMin = Number.NEGATIVE_INFINITY;
429 		event.tMax = Number.POSITIVE_INFINITY;
430 	});
431 	this.segments.forEach(function(segment) {
432 		segment.tStart = undefined;
433 	});
434 };

436 /**
437  * Executes the Critical Path Method.
438  */
439 Storyboard.prototype.executeCPM = function() {
440 	this.topologicalSort();
441 	this.resetTimes();
442 
443 	// Forward
444 	this.events[0].tMin = 0;
445 	for (var i = 1; i < this.events.length; i++) {
446 		var segments = this.events[i].incomingSegments;
447 		for (var j = 0; j < segments.length; j++) {
448 			var from = segments[j].from;
449 			var duration = segments[j].duration;
450 			this.events[i].tMin = Math.max(this.events[i].tMin, from.tMin + duration);
451 		};
452 	};
453 
454 	// Backward
455 	var last = this.events.length - 1;
456 	this.events[last].tMax = this.events[last].tMin;
457 	for (var i = last - 1; i >= 0; i--) {
458 		var segments = this.events[i].outgoingSegments;
459 		for (var j = 0; j < segments.length; j++) {
460 			var to = segments[j].to;
461 			var duration = segments[j].duration;
462 			this.events[i].tMax = Math.min(this.events[i].tMax, to.tMax - duration);
463 		};
464 	};
465 };

490 /**
491  * Returns the structured data the timeline might need to show the segments of the actors
492  * @return {Object[]} Structured data of the segments associated to the actor
493  * @param {Actor[]} actors The actors whose associated segment data must be returned
494  */
495 Storyboard.prototype.actors2SegmentsData = function (actors) {
496 
497 	var ActorSegments = function (actor) {
498 		this.id = actor.id;
499 		this.name = actor.description;
500 		this.series = [];
501 	}
502 	var timelineData = [];
503 	var actorSegmentList = [];
504 	actors.forEach(function (actor) {
505 		var as = new ActorSegments(actor);
506 		actorSegmentList.push(as);
507 	});
508 
509 	this.segments.forEach(function (segment) {
510 		var stop = false;
511 		for(var i=0; i < actorSegmentList.length && !stop; i++) {
512 			if(segment.actor.id === actorSegmentList[i].id){
513 				stop = true;
514 				var bar = {name: segment.id, start: segment.tStart, end: (segment.tStart + segment.duration)};
515 				actorSegmentList[i].series.push(bar);
516 			}
517 		}
518 	});
519 
520 	actorSegmentList.forEach(function (as) {
521 		var al = {id: as.id, name: as.name, series: as.series};
522 		timelineData.push(al);
523 	});
524 	return timelineData;
525 }
